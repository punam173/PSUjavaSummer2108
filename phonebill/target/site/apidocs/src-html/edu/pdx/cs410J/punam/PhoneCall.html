001package edu.pdx.cs410J.punam;
002
003import edu.pdx.cs410J.AbstractPhoneCall;
004
005/**
006 * This class represents a PhoneCall that extends an abstract PhoneCall class
007 * and override all its abstract methods. This class which stores all the information
008 * regarding a call made, like, caller name, callee name, start and end time and date.
009 * <code>PhoneCall</code>
010 * @author Punam Rani pal
011 */
012public class PhoneCall extends AbstractPhoneCall {
013  private String caller = ""; //refers to the caller(customer) name
014  private String callee = ""; //refers to the callee name(person who is being called)
015  private String startDateTime = "";//refers to the time when the call is made
016  private String endDateTime = "";//refers to the time when the call is terminated
017
018  /**
019   * This method set the customet name <code>setCaller</code>
020   */
021  public void setCaller(String Callee){
022    this.callee = Callee;
023  }
024  @Override
025  /**
026   * This method get the customet name <code>getCaller</code>
027   * @return the customer name
028   */
029  public String getCaller() {
030
031    return this.callee;
032  }
033  /**
034   * This method set the callee name <code>setCallee</code>
035   */
036  public void setCallee( String Callee){
037    this.callee = Callee;
038  }
039  @Override
040  /**
041   * This method get the callee name <code>getCallee</code>
042   * @return the callee name
043   */
044  public String getCallee() {
045
046    return this.callee;
047  }
048  /**
049   * This method set the start date and time of a phone call <code>setStartTimeString</code>
050   */
051  public void setStartDateTime( String dateTime){
052    this.startDateTime = dateTime;
053  }
054  @Override
055  /**
056   * This method get the start date and time of a phone call <code>getStartTimeString</code>
057   * @return the start date and time of a call
058   */
059  public String getStartTimeString() {
060
061    return this.startDateTime;
062  }
063  /**
064   * This method set the end date and time of a phone call <code>setEndTimeString</code>
065   */
066  public void setEndDateTime( String dateTime){
067    this.endDateTime = dateTime;
068  }
069  @Override
070  /**
071   * This method get the end date and time of a phone call <code>getEndTimeString</code>
072   * @return the end date and time of a call
073   */
074  public String getEndTimeString() {
075
076    return this.endDateTime;
077  }
078}
