001package edu.pdx.cs410J.punam;
002/**
003 * The main class for the CS410J PhoneBill Project <code>Project1</code>
004 * @author Punam Rani Pal
005 */
006public class Project1 {
007  /**
008   * This method is the main method where the execution for the program begins
009   * its parse the command line arguments and results in the desired output
010   * <code>main</code>
011   * @param args
012   *        this takes command line arguments of type String
013   */
014  public static void main(String[] args) {
015    boolean isPrint = false; //refers to -print option presence/absence
016    boolean isReadMe = false; //refers to -Readme option presence/absence
017    PhoneCall call = new PhoneCall();  // Refer to one of Dave's classes so that we can be sure it is on the classpath
018    int i = 0; //keeps track of the number of argument
019    int flag = 0;//keeps tack of the position of the -print option
020    if(args.length == 8) {
021      i += 1; //refers to the first position(-print/README)
022      if(Validate.isOptionPrint(args[0])) {
023        isPrint = true;
024        if(!Validate.validateArgs(args)) { //checking the arguments correctness
025          Validate.printUsage();
026          System.exit(1);
027        }
028      }else if(Validate.isOptionReadME(args[0])) { //if readme option is ON
029        isReadMe = true;
030        if(isReadMe) {
031          Validate.printReadMe();
032        }
033      }
034    }else if(args.length == 9) { //when all the arguments are entered (9)
035       if(Validate.isOptionPrint(args[0])){ //if -print option is the first argument
036        isPrint = true;
037       }
038       if(Validate.isOptionPrint(args[1])){ //if -print option is the 2nd argument
039         flag = 1;
040         isPrint = true;
041       }
042      if(Validate.isOptionReadME(args[0]) || Validate.isOptionReadME(args[1])) { //if readme option is ON
043        isReadMe = true;
044      }
045
046      i += 2; //sets the first two arguments positions for the options
047    }
048    //if there are 9 arguments
049    if(isPrint && flag == 0) {  //if print option is the first argument then validate arguments and switch to readme
050      if (!Validate.validateArgs(args)) {
051        Validate.printUsage();
052        if(isReadMe)
053           Validate.printReadMe();
054      }
055      //sets the argument based on the position od the 9 arguments
056      call.setCaller(args[1 + i]);
057      call.setCallee(args[2 + i]);
058      call.setStartDateTime(args[3 + i] + " " + args[4 + i]);
059      call.setEndDateTime(args[5 + i] + " " + args[6 + i]);
060
061      PhoneBill bill = new PhoneBill(); //creates a new phonebill for the customer
062      bill.setCustomer(args[0 + i]); //sets the customer name from the argument on the basis of number of arguments presents.
063      bill.addPhoneCall(call); //add the new call information to the phonebill collection
064      //if all the arguments are correct and if print is the first one then -print prints out first
065      //then swtch to readme else just stwitch to README.
066      if (isPrint) {
067        for (PhoneCall pc : bill.getPhoneCalls()) {
068          System.out.println(pc);
069        }
070        System.out.println(bill);
071      }
072      if (isReadMe)
073        Validate.printReadMe();
074    }
075    if(isReadMe && flag == 1)
076       Validate.printReadMe();
077
078  }
079}
