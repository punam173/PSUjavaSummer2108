003/**
004 * This class represents a <code>Validate</code>

006 */
007public class Validate {
008    /**
009     * This method validates the arguments passed in the command line <code>validateArgs</code>
010     * @param args
011     *        takes an array of arguments the maximum size if 9
012     * @return boolean result
013     *         false if the arguments are invalid
014     *         true if teh arguments are valid
015     */
016    public static boolean validateArgs(String[] args) {
017
018        if (args == null || args.length == 0 || args.length < 7 || args.length > 9) {
019            System.out.println("Invalid Arguments");
020            return false;
021        }
022        if (args.length == 8) {
023            if(!isOptionPrint(args[0]) && !isOptionReadME(args[0])) {
024                return false;
025            }
026        } else if (args.length == 9) {
027            if((!isOptionPrint(args[0]) &&  !isOptionReadME(args[0])) ||
028                    (!isOptionPrint(args[1]) && !isOptionReadME(args[1]))) {
029                return false;
030            }
031        }
032        //Initialize assuming there are no options
033        String customer = args[0];
034        String caller = args[1];
035        String callee = args[2];
036        String startDate = args[3];
037        String startTime = args[4];
038        String endDate = args[5];
039        String endTime = args[6];
040        if (args.length == 9) {
041            customer = args[2];
042            caller = args[3];
043            callee = args[4];
044            startDate = args[5];
045            startTime = args[6];
046            endDate = args[7];
047            endTime = args[8];
048        } else if (args.length == 8) {
049            customer = args[1];
050            caller = args[2];
051            callee = args[3];
052            startDate = args[4];
053            startTime = args[5];
054            endDate = args[6];
055            endTime = args[7];
056        }
057        //regex pattern for alpha-numeric
058        if (!customer.matches("[a-zA-Z0-9\\s]+")) {
059            System.out.println("Invalid customer name format");
060            return false;
061        }
062
063        if (!caller.matches("\\d{3}[-]\\d{3}[-]\\d{4}")) {
064            System.out.println("Invalid caller format");
065            return false;
066        }
067
068        if (!callee.matches("\\d{3}[-]\\d{3}[-]\\d{4}")) {
069            System.out.println("Invalid callee format");
070            return false;
071        }
072
073        if(!startDate.matches("\\d{1,2}[/]\\d{1,2}[/]\\d{4}")) {
074            System.out.println("Invalid Date Format");
075            return false;
076        }
077
078        if(!startTime.matches("\\d{2}[:]\\d{2}")) {
079            System.out.println("Invalid Time format");
080            return false;
081        }
082
083        if(!endDate.matches("\\d{1,2}[/]\\d{1,2}[/]\\d{4}")) {
084            System.out.println("Invalid Date Format");
085            return false;
086        }
087
088        if(!endTime.matches("\\d{2}[:]\\d{2}")) {
089            System.out.println("Invalid Time format");
090            return false;
091        }
092
093        return true;
094    }
095    /**
096     * This method checks for the -print option <code>isOptionPrint</code>
097     * @param option
098     *        takes the either of first two arguents depending on the size.
099     * @return boolean
100     *         returns true if the option is -print else false
101     */
102    public static boolean isOptionPrint(String option) {
103        if(option != null && option.startsWith("-")) {
104            if(!option.equals("-print")) {
105                return false;
106            }
107        }
108        return true;
109    }
110    /**
111     * This method hecks if the option is -README <code>isOptionReadme</code>
112     * @param option
113     *        either of the first two arguents
114     * @return boolean
115     *         true if the option is -README else false
116     */
117    public static boolean isOptionReadME(String option) {
118        if(option != null && option.startsWith("-")) {
119            if(!option.equals("-README")) {
120                return false;
121            }
122        }
123        return true;
124    }
125    /**

127     */

130        System.out.println("args are (in this order): ");
131        System.out.println("1: customer Person whose phone bill weâ€™re modeling");
132        System.out.println("2: calleeNumber Phone number of person who was called");
133        System.out.println("3: startTime Date and time call began (24-hour time)");
134        System.out.println("4: endTime Date and time call ended (24-hour time)");
135        System.out.println("options are (options may appear in any order): ");
136        System.out.println("5: -print Prints a description of the new phone call");
137        System.out.println("6: -README Prints a README for this project and exits");
138        System.out.println("7: Date and time should be in the format: mm/dd/yyyy hh:mm");
139    }
140    /**
141     * This method prints the README <code>printReadMe</code>
142     */

144        System.out.println("README*****************\n" +
145                "Project Name: Phonebill\n" + "Programmed By: Punam Rani Pal\n" +
